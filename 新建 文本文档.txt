<script>
    (function() {
        // DOM 元素
        const mixSlider = document.getElementById('mixSlider');
        const moistSlider = document.getElementById('moistSlider');
        const heatSlider = document.getElementById('heatSlider');
        const mixVal = document.getElementById('mixVal');
        const moistVal = document.getElementById('moistVal');
        const heatVal = document.getElementById('heatVal');
        const volatileInput = document.getElementById('volatile');
        const ashInput = document.getElementById('ash');
        const citySelect = document.getElementById('citySelect');
        const wasteSelect = document.getElementById('wasteTypeSelect');

        const preEff = document.getElementById('preEff');
        const preNOx = document.getElementById('preNOx');
        const preCO2 = document.getElementById('preCO2');
        const preCost = document.getElementById('preCost');
        const optRatio = document.getElementById('optRatio');

        // 图表实例
        let pollChart = echarts.init(document.getElementById('pollChart'));
        let pareChart = echarts.init(document.getElementById('pareChart'));

        // 默认图表数据（静态示例）
        function initCharts() {
            pollChart.setOption({
                tooltip: { trigger: 'axis' },
                legend: { bottom: 5, data: ['NOx', 'SO₂'] },
                grid: { left: '12%', right: '8%', top: '15%', bottom: '20%' },
                xAxis: { type: 'category', data: ['0%', '4%', '8%', '12%', '16%', '20%'], name: '掺烧比例' },
                yAxis: { type: 'value', name: 'mg/m³', min: 0, max: 250 },
                series: [
                    { name: 'NOx', type: 'line', data: [160, 168, 175, 186, 198, 215], color: '#2f855a' },
                    { name: 'SO₂', type: 'line', data: [200, 205, 212, 220, 232, 245], color: '#d97706' }
                ]
            });
            let data = [];
            for (let i=0;i<30;i++) data.push([200+Math.random()*40, 140+Math.random()*50]);
            pareChart.setOption({
                tooltip: {},
                grid: { left: '12%', right: '8%', top: '10%', bottom: '15%' },
                xAxis: { type: 'value', name: '运行成本 (元/吨)' },
                yAxis: { type: 'value', name: 'NOx排放 (mg/m³)' },
                series: [{ type: 'scatter', data: data, color: '#2f855a' }]
            });
        }
        initCharts();

        // ---------- 核心函数：更新预测 ----------
        async function updatePredictions() {
            // 收集当前参数
            const params = {
                mix_ratio: parseFloat(mixSlider.value),
                moisture: parseFloat(moistSlider.value),
                heat_value: parseFloat(heatSlider.value),
                volatile: parseFloat(volatileInput.value) || 38,
                ash: parseFloat(ashInput.value) || 32,
                city: citySelect.value,
                waste_type: wasteSelect.value
            };

            // 更新显示滑块数值
            mixVal.innerText = params.mix_ratio + '%';
            moistVal.innerText = params.moisture + '%';
            heatVal.innerText = params.heat_value + ' kJ/kg';

            try {
                // 尝试调用后端API（如果后端未启动，会失败并进入catch）
                const response = await fetch('http://127.0.0.1:5000/predict', {
                    method: 'POST',
                    headers: { 'Content-Type': 'application/json' },
                    body: JSON.stringify(params)
                });
                const result = await response.json();
                if (result.success) {
                    // 用后端返回的真实数据更新界面
                    preEff.innerText = result.boiler_efficiency || '91.3';
                    preNOx.innerText = result.nox_emission || '175';
                    preCO2.innerText = result.co2_emission || '0.82';
                    preCost.innerText = result.operating_cost || '236.5';
                    optRatio.innerText = (result.recommended_ratio || '8.2') + '%';
                    // 可选：更新图表数据
                    return;
                }
            } catch (error) {
                console.log('后端未连接，使用模拟数据');
                // 如果API调用失败，回退到模拟计算（仅用于前端演示）
                fallbackCalculation(params);
            }
        }

        // 模拟计算函数（当后端未启用时使用）
        function fallbackCalculation(p) {
            let mix = p.mix_ratio, moist = p.moisture, heat = p.heat_value;
            // 模拟公式
            let eff = 93.5 - 0.14 * mix - 0.06 * (moist - 30) * (mix/6) - 0.002 * (3000 - heat) * (mix/8);
            eff = Math.min(95, Math.max(86, eff)).toFixed(1);
            let nox = 160 + 2.5 * mix + 0.7 * (moist - 30) - 0.03 * (heat - 2800);
            nox = Math.min(280, Math.max(130, nox)).toFixed(0);
            let co2 = (0.78 + 0.012 * mix - 0.0001 * (heat - 2500) + 0.002 * (moist - 30)).toFixed(2);
            let cost = (220 + 2.2 * mix + 0.8 * (moist - 25) - 0.05 * (heat - 2500)).toFixed(1);
            let opt = (7.5 + 0.02 * (heat - 2800)/10 - 0.1 * (moist - 32)).toFixed(1);
            preEff.innerText = eff;
            preNOx.innerText = nox;
            preCO2.innerText = co2;
            preCost.innerText = cost;
            optRatio.innerText = opt + '%';
        }

        // 绑定事件
        [mixSlider, moistSlider, heatSlider, volatileInput, ashInput, citySelect, wasteSelect].forEach(el => {
            el.addEventListener('input', updatePredictions);
            el.addEventListener('change', updatePredictions);
        });

        // 重置按钮
        document.getElementById('resetBtn').addEventListener('click', function() {
            mixSlider.value = 8.2;
            moistSlider.value = 34;
            heatSlider.value = 2950;
            volatileInput.value = 38;
            ashInput.value = 32;
            citySelect.value = '上海';
            wasteSelect.value = '市政污泥';
            updatePredictions();
        });

        // 执行优化按钮（模拟触发优化，可调用后端优化接口）
        document.getElementById('runOptimBtn').addEventListener('click', function() {
            alert('优化请求已发送，请等待后端返回结果（目前演示模式）。');
            updatePredictions(); // 重新调用更新，实际应调用优化API
        });

        // 上传文件模拟
        document.getElementById('uploadTrigger').addEventListener('click', function() {
            alert('实际应用中会上传文件并解析参数。');
        });

        // 初始调用
        updatePredictions();
    })();
</script>
